style-dictionary.config.jsconfig
export default {
  source: ["src/design-tokens/**/*.json"],
  platforms: {
    css: {
      transformGroup: "css",
      buildPath: "assets/styles/tokens/",
      files: [{
        destination: "primitive.css",
        format: "css/variables"
      }]
    }
  }
}
