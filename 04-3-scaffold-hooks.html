import { useState, useEffect } from 'react';export function UserList() { const [data, setData] = useState([]); ① 상태 3종 수동 선언 const [loading, setLoading] = useState(true); const [error, setError] = useState(null); useEffect(() => { ② fetch·로딩·에러 직접 처리 fetch('/api/users') .then(r => { if (!r.ok) throw Error(r.status); // 상태코드 에러 return r.json(); }) .then(setData).catch(setError) .finally(() => setLoading(false)); }, []); ③ 캐시·갱신 직접 구현 if (loading) return <Spinner/>; return <Table rows={data}/>;}
