Accordion 컴포넌트

@axiom/components/ui 라이브 예제

기본 저장 공간, API 접근, 이메일 지원이 포함됩니다. 개인 프로젝트에 적합합니다.
우선 지원, 팀 협업 기능, 고급 분석 도구가 함께 제공됩니다.
전담 지원, 무제한 저장 공간, SLA 보장, 커스텀 연동을 제공합니다.
현재 열린 항목 : 없음
AccordionExample.tsx 
import { useState } from 'react';
import {
  Accordion, AccordionItem,
  AccordionTrigger, AccordionContent,
} from '@axiom/components/ui';

export default function Example() {
  const [open, setOpen] = useState('item-1');

  return (
    <Accordion type="single" collapsible
      value={open} onValueChange={setOpen}>
      <AccordionItem value="item-1">
        <AccordionTrigger>스타터 플랜 ($9/월)</AccordionTrigger>
        <AccordionContent>기본 저장 공간, API 접근…</AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-2">
        <AccordionTrigger>프로 플랜 ($29/월)</AccordionTrigger>
        <AccordionContent>우선 지원, 팀 협업…</AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-3">
        <AccordionTrigger>엔터프라이즈 ($99/월)</AccordionTrigger>
        <AccordionContent>전담 지원, 무제한 저장…</AccordionContent>
      </AccordionItem>
    </Accordion>
  );
}
